using Distributed
@everywhere begin
    using PyCall
    cma = pyimport("cma")
    using DataFrames
    function BenchmarkCMA1(fi, ftol, dim, i)
        fitness = cma.bbobbenchmarks.dictbbob[fi](i)
        es = cma.CMAEvolutionStrategy(zeros(dim), 3, 
            Dict("verb_log"=>0,"verb_disp"=>0,"maxfevals"=>10000*dim))
        out = es.minimize(fitness).result
        df = out[2]- fitness.getfopt()
        evals = out[4]
        DataFrame(success = df<=ftol, evals = evals, dim=dim, i = i, fitness = "F$fi")
    end

end
# BenchmarkCMA1(9, 1e-8, 5, 1)
@everywhere using ProgressMeter
using Statistics
function BenchmarkCMA(args1...)
    out = @showprogress pmap(args->BenchmarkCMA1(args...), 
            Iterators.product(args1...))
    out = reduce(vcat, vec(out))
    return out
end

using Distributed
@everywhere begin
    include("QGA2.jl")
    using DataFrames
    using PyCall
    cma = pyimport("cma")
    function BenchmarkQGA1(fi, ftol, dim, s, mumin, i)
        fitness = cma.bbobbenchmarks.dictbbob[fi](i)
        out = QGA.minimize(fitness, zeros(dim), 3*ones(dim), s,
            MaxEvals = 10000*dim, 
            FminTarget = fitness.getfopt()+ftol,
            mumin = mumin)
        DataFrame(halt = out[:halt], evals = out[:evals], success = out[:halt] == :FminTarget,
            dim=dim, i = i, s=s, mumin = mumin, fitness = "F$fi")
    end
end
@everywhere using ProgressMeter
using Statistics
function BenchmarkQGA(args1...)
    out = @showprogress pmap(args->BenchmarkQGA1(args...), 
            Iterators.product(args1...))
    out = reduce(vcat, vec(out))
    return out
end
